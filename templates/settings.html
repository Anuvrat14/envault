Quick Start
# 1. Download the CLI script
curl -o vaultic {{ request.url_root }}cli/download && chmod +x vaultic
# 2. Install globally
sudo mv vaultic /usr/local/bin/vaultic
# 3. Save your token
mkdir -p ~/.vaultic && mv ~/Downloads/vaultic_cli_token ~/.vaultic/cli_token && chmod 600 ~/.vaultic/cli_token
# 4. Use it
vaultic inject "My App" -- npm run dev
