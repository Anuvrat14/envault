Quick Start
# 1. Download the CLI script
curl -o dotward {{ request.url_root }}cli/download && chmod +x dotward
# 2. Install globally
sudo mv dotward /usr/local/bin/dotward
# 3. Save your token
mkdir -p ~/.dotward && mv ~/Downloads/dotward_cli_token ~/.dotward/cli_token && chmod 600 ~/.dotward/cli_token
# 4. Use it
dotward inject "My App" -- npm run dev
